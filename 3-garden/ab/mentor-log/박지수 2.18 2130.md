```tsx
function useAsync<Response , Params >({
fetchApi,
onError
}: {
  fetchApi: (params: Params) => Promise<Response>;
  onError?: (error: Error) => void;
}) {
  const [isFetching, setIsFetching] = useState(false)
  const [data, setData] = useState<Response>()
  const asyncFunction = async (params: Params) => {
    setIsFetching(true)
    try {
      const response = await fetchApi(params)
      setData(response);
      return response;
    } catch (error) {
      onError(error)
    } finally {
      setIsFetching(false)
    }
  }

  return {asyncFunction, isFetching, data}
}
```
