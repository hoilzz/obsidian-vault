```ts
export type JwtClaim = {
  authToken?: AuthToken;
  kakaoAuthToken?: KakaoAuthToken;

  user?: User;
  appConfig?: AppConfig;
  /**
   * 필수 약관 동의 진행 필요 여부
   */
  requireAcceptAgreement?: boolean;
  /*
   * 선택 약관 동의 진행 필요 여부
   */
  optionalAcceptAgreement?: boolean;
  /**
   * JWT 토큰의 최종 만료 일자.
   * 기본적으로 JWT 토큰은 하루 만료로, refresh_token 마다 하루씩 연장됩니다. maxExpireDate 는 연장되는 토큰의 최종 만료 일자입니다. 여기에 도달하게 되면 더 이상 연장되지 않습니다.
   * 참고로 refresh_token 을 하기 위해선 카카오SDK 에서 제공하는 새로운 access_token 을 전달해야하기 때문에 만약 갱신된 엑세스 토큰이 전달되지 않으면 refresh_token 도 되지 않습니다
   * 그래서 maxExpireDate 를 최대한 길게 잡아도 보안 이슈는 없을 것으로 보입니다. (검증된 access_token 은 카카오SDK 에서 알아서 전달해주기 때문)
   */
  maxExpireDate?: string;
};
```
